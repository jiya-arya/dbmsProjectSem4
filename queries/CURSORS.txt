-- 1 declare cursor to fetch all employees and display names
DECLARE
    CURSOR c1 IS SELECT ENAME FROM EMP;
    name_var EMP.ENAME%TYPE;
BEGIN
    OPEN c1;
    LOOP
        FETCH c1 INTO name_var;
        EXIT WHEN c1%NOTFOUND;
        DBMS_OUTPUT.PUT_LINE(name_var);
    END LOOP;
    CLOSE c1;
END;
/

-- 2 fetch employee name and salary using cursor
DECLARE
    CURSOR c1 IS SELECT ENAME, SAL FROM EMP;
    name_var EMP.ENAME%TYPE;
    sal_var EMP.SAL%TYPE;
BEGIN
    OPEN c1;
    LOOP
        FETCH c1 INTO name_var, sal_var;
        EXIT WHEN c1%NOTFOUND;
        DBMS_OUTPUT.PUT_LINE(name_var || ' ' || sal_var);
    END LOOP;
    CLOSE c1;
END;
/

/

-- 4 count number of employees using cursor
DECLARE
    CURSOR c1 IS SELECT EMPNO FROM EMP;
    count_var NUMBER := 0;
    temp EMP.EMPNO%TYPE;
BEGIN
    OPEN c1;
    LOOP
        FETCH c1 INTO temp;
        EXIT WHEN c1%NOTFOUND;
        count_var := count_var + 1;
    END LOOP;
    CLOSE c1;
    DBMS_OUTPUT.PUT_LINE(count_var);
END;
/

-- 5 calculate total salary using cursor
DECLARE
    CURSOR c1 IS SELECT SAL FROM EMP;
    total NUMBER := 0;
    sal_var EMP.SAL%TYPE;
BEGIN
    OPEN c1;
    LOOP
        FETCH c1 INTO sal_var;
        EXIT WHEN c1%NOTFOUND;
        total := total + sal_var;
    END LOOP;
    CLOSE c1;
    DBMS_OUTPUT.PUT_LINE(total);
END;
/

-- 6 fetch records using OPEN FETCH CLOSE
DECLARE
    CURSOR c1 IS SELECT ENAME FROM EMP;
    name_var EMP.ENAME%TYPE;
BEGIN
    OPEN c1;
    FETCH c1 INTO name_var;
    WHILE c1%FOUND LOOP
        DBMS_OUTPUT.PUT_LINE(name_var);
        FETCH c1 INTO name_var;
    END LOOP;
    CLOSE c1;
END;
/

-- 7 display all employee records using cursor FOR LOOP
BEGIN
    FOR rec IN (SELECT * FROM EMP) LOOP
        DBMS_OUTPUT.PUT_LINE(rec.ENAME || ' ' || rec.SAL);
    END LOOP;
END;
/

-- 8 display employee names and jobs using cursor FOR LOOP
BEGIN
    FOR rec IN (SELECT ENAME, JOB FROM EMP) LOOP
        DBMS_OUTPUT.PUT_LINE(rec.ENAME || ' ' || rec.JOB);
    END LOOP;
END;
/

-- 9 display employees from department 10
BEGIN
    FOR rec IN (SELECT * FROM EMP WHERE DEPTNO = 10) LOOP
        DBMS_OUTPUT.PUT_LINE(rec.ENAME);
    END LOOP;
END;
/

-- 10 calculate total salary using cursor FOR LOOP
DECLARE
    total NUMBER := 0;
BEGIN
    FOR rec IN (SELECT SAL FROM EMP) LOOP
        total := total + rec.SAL;
    END LOOP;
    DBMS_OUTPUT.PUT_LINE(total);
END;
/

-- 11 employees with salary > average
DECLARE
    avg_sal NUMBER;
BEGIN
    SELECT AVG(SAL) INTO avg_sal FROM EMP;

    FOR rec IN (SELECT ENAME, SAL FROM EMP WHERE SAL > avg_sal) LOOP
        DBMS_OUTPUT.PUT_LINE(rec.ENAME || ' ' || rec.SAL);
    END LOOP;
END;
/

-- 12 parameterized cursor for department
DECLARE
    CURSOR c1(dno NUMBER) IS 
        SELECT ENAME FROM EMP WHERE DEPTNO = dno;
    name_var EMP.ENAME%TYPE;
BEGIN
    OPEN c1(10);
    LOOP
        FETCH c1 INTO name_var;
        EXIT WHEN c1%NOTFOUND;
        DBMS_OUTPUT.PUT_LINE(name_var);
    END LOOP;
    CLOSE c1;
END;
