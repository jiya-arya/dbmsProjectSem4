-- 1 check if a number is positive
DECLARE
    n NUMBER := 10;
BEGIN
    IF n > 0 THEN
        DBMS_OUTPUT.PUT_LINE('Positive');
    END IF;
END;

-- 3 check if number is even

-- 4 display message if today is Monday
BEGIN

-- 11 assign grades using IF-ELSIF
DECLARE
    marks NUMBER := 75;
BEGIN
    IF marks >= 80 THEN
        DBMS_OUTPUT.PUT_LINE('A');
    ELSIF marks >= 60 THEN
        DBMS_OUTPUT.PUT_LINE('B');
    ELSIF marks >= 40 THEN
        DBMS_OUTPUT.PUT_LINE('C');
    ELSE
        DBMS_OUTPUT.PUT_LINE('Fail');
    END IF;
END;
/

-- 12 greatest of three numbers
DECLARE
    a NUMBER := 10;
    b NUMBER := 20;
    c NUMBER := 15;
BEGIN

        DBMS_OUTPUT.PUT_LINE(a);
    ELSIF b > c THEN
        DBMS_OUTPUT.PUT_LINE(b);
    ELSE
        DBMS_OUTPUT.PUT_LINE(c);
    END IF;
END;
/

-- 16 CASE day of week
DECLARE
    d NUMBER := 3;
BEGIN
    CASE d
        WHEN 1 THEN DBMS_OUTPUT.PUT_LINE('Monday');
        WHEN 2 THEN DBMS_OUTPUT.PUT_LINE('Tuesday');
        WHEN 3 THEN DBMS_OUTPUT.PUT_LINE('Wednesday');
        ELSE DBMS_OUTPUT.PUT_LINE('Other');
    END CASE;
END;

-- 26 exit when sum exceeds 50
DECLARE
    sum_val NUMBER := 0;
    i NUMBER := 1;
BEGIN
    LOOP
        sum_val := sum_val + i;
        EXIT WHEN sum_val > 50;
        i := i + 1;
    END LOOP;
    DBMS_OUTPUT.PUT_LINE(sum_val);
END;
/

-- 28 exit when condition met
DECLARE
    i NUMBER := 1;
BEGIN
    LOOP
        EXIT WHEN i > 5;
        DBMS_OUTPUT.PUT_LINE(i);
        i := i + 1;
    END LOOP;
END;
/

-- 31 simple login system

    uname VARCHAR2(20) := 'admin';
    pass VARCHAR2(20) := '123';

        DBMS_OUTPUT.PUT_LINE('Login Success');
    ELSE
        DBMS_OUTPUT.PUT_LINE('Login Failed');
    END IF;
END;
