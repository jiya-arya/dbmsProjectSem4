-- 24 fetch employee with highest salary
DECLARE
    emp_rec EMP%ROWTYPE;
BEGIN
    SELECT * INTO emp_rec FROM EMP
    WHERE SAL = (SELECT MAX(SAL) FROM EMP);
    DBMS_OUTPUT.PUT_LINE(emp_rec.ENAME);
END;
/

-- 25 display employees of a department using loop
DECLARE
BEGIN
    FOR rec IN (SELECT * FROM EMP WHERE DEPTNO = 10) LOOP
        DBMS_OUTPUT.PUT_LINE(rec.ENAME);
    END LOOP;
END;
/

-- 26 increase salary department-wise
BEGIN
    FOR rec IN (SELECT EMPNO FROM EMP WHERE DEPTNO = 10) LOOP
        UPDATE EMP SET SAL = SAL + 200 WHERE EMPNO = rec.EMPNO;
    END LOOP;
END;
/

-- 27 cursor with %ROWTYPE
DECLARE
    CURSOR c1 IS SELECT * FROM EMP;
    emp_rec EMP%ROWTYPE;
BEGIN
    OPEN c1;
    LOOP
        FETCH c1 INTO emp_rec;
        EXIT WHEN c1%NOTFOUND;
        DBMS_OUTPUT.PUT_LINE(emp_rec.ENAME);
    END LOOP;
    CLOSE c1;
END;
/

-- 28 copy data using %ROWTYPE
DECLARE
    emp_rec EMP%ROWTYPE;
BEGIN
    FOR rec IN (SELECT * FROM EMP) LOOP
        INSERT INTO EMP_COPY VALUES rec;
    END LOOP;
END;
