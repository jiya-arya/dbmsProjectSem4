-- 4 display employee name and salary using variables
DECLARE
    ename_var EMP.ENAME%TYPE;
    sal_var EMP.SAL%TYPE;
BEGIN
    SELECT ENAME, SAL INTO ename_var, sal_var
    FROM EMP
    WHERE EMPNO = 4300;

    DBMS_OUTPUT.PUT_LINE(ename_var || ' ' || sal_var);
END;
/

-- 7 store employee name and salary and display
DECLARE
    name_var EMP.ENAME%TYPE;
    salary_var EMP.SAL%TYPE;
BEGIN
    SELECT ENAME, SAL INTO name_var, salary_var
    FROM EMP
    WHERE EMPNO = 7499;

    DBMS_OUTPUT.PUT_LINE(name_var || ' ' || salary_var);
END;
/

-- 8 swap two numbers
DECLARE
    a NUMBER := 10;
    b NUMBER := 20;
    temp NUMBER;
BEGIN
    temp := a;
    a := b;
    b := temp;

    DBMS_OUTPUT.PUT_LINE('A=' || a || ' B=' || b);
END;
/

-- 9 calculate area of rectangle
DECLARE
    l NUMBER := 5;
    w NUMBER := 4;
BEGIN
    DBMS_OUTPUT.PUT_LINE('Area=' || (l*w));
END;
/

-- 10 calculate area of circle using constant PI
DECLARE
    r NUMBER := 7;
    pi CONSTANT NUMBER := 3.14;
BEGIN
    DBMS_OUTPUT.PUT_LINE('Area=' || (pi*r*r));
END;
/

-- 11 calculate salary after tax
DECLARE
    sal NUMBER := 5000;
    tax CONSTANT NUMBER := 0.1;
BEGIN
    DBMS_OUTPUT.PUT_LINE('After Tax=' || (sal - sal*tax));
END;
/

-- 13 simple interest using constant rate
DECLARE
    p NUMBER := 1000;
    r CONSTANT NUMBER := 0.05;
    t NUMBER := 2;
BEGIN
    DBMS_OUTPUT.PUT_LINE('SI=' || (p*r*t));
END;
/

-- 14 using different datatypes

/

-- 15 declare variable using %TYPE
DECLARE
    sal_var EMP.SAL%TYPE;
BEGIN
    SELECT SAL INTO sal_var FROM EMP WHERE EMPNO = 7369;
    DBMS_OUTPUT.PUT_LINE(sal_var);
END;
/

-- 16 declare record using %ROWTYPE
DECLARE
    emp_rec EMP%ROWTYPE;
BEGIN
    SELECT * INTO emp_rec FROM EMP WHERE EMPNO = 7369;
    DBMS_OUTPUT.PUT_LINE(emp_rec.ENAME || ' ' || emp_rec.SAL);
END;
/
